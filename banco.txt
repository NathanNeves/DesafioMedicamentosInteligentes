CREATE TABLE Usuario (
    email varchar(255),
    nome VARCHAR(50),
    data_de_Nascimento DATE,
    telefone VARCHAR(16),
    id SERIAL PRIMARY KEY
);

CREATE TABLE Carrinho (
    id SERIAL PRIMARY KEY,
    quantidade INT,
    fk_Usuario_Id int,
    fk_Produtos_Id INT
);

CREATE TABLE Endereco (
    enderecoId Serial PRIMARY KEY,
    logradouro VARCHAR(255),
    localidade VARCHAR(100),
    cep VARCHAR(10),
    fk_Usuario_Id int
);

CREATE TABLE Produtos (
    nome VARCHAR(50),
    codProduto VARCHAR(255),
    id Serial PRIMARY KEY,
    quantidade INT
);

CREATE TABLE Compra (
    fk_Usuario_Id int,
    fk_Produtos_Id INT,
    quantidade BIGINT,
    transacao boolean,
    entregue boolean,
    dataentrega Date
);
